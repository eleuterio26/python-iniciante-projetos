pacientes = []

while True:
    print("\n=== SISTEMA DE PACIENTES ===")
    print("1 - Cadastrar paciente")
    print("2 - Listar pacientes")
    print("3 - Buscar paciente")
    print("4 - Sair")

    opcao = input("Escolha uma opção: ")

    if opcao == "1":
        nome = input("Nome: ")
        idade = input("Idade: ")
        exame = input("Exame: ")

        paciente = {
            "nome": nome,
            "idade": idade,
            "exame": exame
        }

        pacientes.append(paciente)
        print("Paciente cadastrado com sucesso!")

    elif opcao == "2":
        print("\nLista de pacientes:")
        for p in pacientes:
            print(f"Nome: {p['nome']} | Idade: {p['idade']} | Exame: {p['exame']}")

    elif opcao == "3":
        busca = input("Digite o nome do paciente: ")
        encontrado = False

        for p in pacientes:
            if p["nome"].lower() == busca.lower():
                print(f"Encontrado: {p}")
                encontrado = True

        if not encontrado:
            print("Paciente não encontrado.")

    elif opcao == "4":
        print("Saindo do sistema...")
        break

    else:
        print("Opção inválida!")
